```yaml
name: [网站名称]
    link: [网站链接]
    avatar: [网站logo链接]
    location: [您的所在地，可选]
    organization: [您的所属组织，可选]
    desc: [您的个性签名\口号]
    socials:[您的社交账号信息，包括名称与地址，可选可增添]

        icon: github
        link: https://github.com/MrSibe
```
